<?php

class Foo {

    /**
     * @inheritDoc
     */
    public function hasIdentity($id): bool
    {
        $identityId = $this->getIdentityId($id);

        return $this->getIdentityMap()->has($identityId);
    }





    /**
     * @inheritDoc
     */
    public function loadFromIdentityMap($id): mixed
    {
        $identityId = $this->getIdentityId($id);

        return $this->getIdentityMap()->get($identityId);
    }





    /**
     * @inheritDoc
     */
    public function getIdentityId($id): string
    {
        $this->makeSureHasMappedClass();

        return $this->getIdentityMap()->generateId(
            $this->mappedClass,
            $id
        );
    }





    /**
     * @param object $object
     * @return $this
     */
    private function mapClassFromObject(object $object): static
    {
        return $this->mappedClass($object);
    }





    /**
     * @return void
     * @throws NotFoundMappedClass
     */
    private function makeSureHasMappedClass(): void
    {
        if (!$this->mappedClass) {
            throw new NotFoundMappedClass();
        }
    }
}
